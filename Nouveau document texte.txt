CREATE TABLE DatabaseFiches.entite(
	nom VARCHAR(100),
	type VARCHAR(100)
);


CREATE TABLE DatabaseFiches.vehicules(
	marque VARCHAR(100),
	modele VARCHAR(100),
	puissance VARCHAR(40),
	annee INT,
	immatriculation VARCHAR(50)
);


CREATE TABLE DatabaseFiches.individus(
	nom VARCHAR(100),
	prenom VARCHAR(100),
	fonction VARCHAR(255)
);

CREATE TABLE DatabaseFiches.fiches(
	date DATE,
	compteurDepart INT,
	compteurArrivee INT,
	lieuDepart VARCHAR(255),
	lieuArrivee VARCHAR(255),
	commentaire TEXT,
	objet TEXT,
	idEntite,
	idIndividus,
	idVehicule
);
